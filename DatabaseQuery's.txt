Database Script

Database Name : CarParkingSystem

Commands :
Create database CarParkingSystem;

Use database CarParkingSystem;

Table Users {
  ID integer [primary key]
  CYGID varchar [unique]
  Name varchar
  PhoneNumber varchar [unique]
  Email varchar [unique]
  Password varchar
  Role bool
  CreatedBy varchar
  CreatedAt timestamp
  UpdatedBy varchar
  UpdatedAt timestamp
  CarID varchar  // Foreign Key referencing Cars.CarID
}
 
Table Cars {
  ID varchar [primary key]
  CarNumber varchar
  CarModel varchar
  StatusID varchar  // Foreign Key referencing CarStatus.StatusID
  CreatedBy varchar
  CreatedAt timestamp
  UpdatedBy varchar
  UpdatedAt timestamp
  OwnerID varchar  // Foreign Key referencing Users.UserID
  ValetID varchar  // Foreign Key referencing Users.UserID
}
 
Table CarStatus {
  ID varchar [primary key]
  CarID varchar  // Foreign Key referencing Cars.CarID
  Status varchar
}
 
Table CarStatusLog {
  ID integer [primary key]
  UserID varchar  // Foreign Key referencing Users.UserID
  CarID varchar  // Foreign Key referencing Cars.CarID
  StatusID varchar
  CreatedAt timestamp
  UpdatedAt timestamp
}
 
// Foreign Key Constraints
Ref: Users.CarID > Cars.ID
Ref: Cars.OwnerID > Users.ID
Ref: Cars.ValetID > Users.ID
Ref: Cars.StatusID > CarStatus.ID
Ref: CarStatus.CarID > Cars.ID
Ref: CarStatusLog.UserID > Users.ID
Ref: CarStatusLog.CarID > Cars.ID
Ref: CarStatusLog.StatusID > CarStatus.ID
